3-Step Installation Guide (Required for cross-device syncing)

  1. Create a new Google Sheet (go to sheets.new). Go to Extensions > Apps Script.
  2. Delete any existing code and copy-paste the snippet provided below into the Apps Script editor.
  3. Click Deploy > New deployment, select type Web App, change "Who has access" to "Anyone", click deploy, and paste the URL above!
function doPost(e) {
  try {
    var payload = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    
    // Ensure Users sheet exists
    var usersSheet = ss.getSheetByName("Users");
    if (!usersSheet) {
      usersSheet = ss.insertSheet("Users");
      usersSheet.appendRow(["Email", "Password"]);
    }
    
    // Ensure Timesheets sheet exists
    var tsSheet = ss.getSheetByName("Timesheets");
    if (!tsSheet) {
      tsSheet = ss.insertSheet("Timesheets");
      tsSheet.appendRow(["Email", "WeekID", "DataJSON", "LastUpdated"]);
    }
    
    var action = payload.action;
    
    if (action === "register") {
      var email = payload.email.toLowerCase();
      var password = payload.password;
      
      // Check if user exists
      var data = usersSheet.getDataRange().getValues();
      for (var i = 1; i < data.length; i++) {
        if (data[i][0].toLowerCase() === email) {
          return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "User already exists" }))
            .setMimeType(ContentService.MimeType.JSON);
        }
      }
      
      usersSheet.appendRow([email, password]);
      return ContentService.createTextOutput(JSON.stringify({ status: "success", message: "User registered" }))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
    if (action === "login") {
      var email = payload.email.toLowerCase();
      var password = payload.password;
      
      var data = usersSheet.getDataRange().getValues();
      for (var i = 1; i < data.length; i++) {
        if (data[i][0].toLowerCase() === email && String(data[i][1]) === String(password)) {
          return ContentService.createTextOutput(JSON.stringify({ status: "success", uid: "usr_" + Utilities.base64Encode(email) }))
            .setMimeType(ContentService.MimeType.JSON);
        }
      }
      return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "Invalid email or password" }))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
    if (action === "save") {
      var email = payload.email.toLowerCase();
      var weekId = payload.weekId;
      var weekDataStr = payload.weekData; 
      
      var data = tsSheet.getDataRange().getValues();
      var foundRow = -1;
      for (var i = 1; i < data.length; i++) {
        if (data[i][0].toLowerCase() === email && data[i][1] === weekId) {
          foundRow = i + 1;
          break;
        }
      }
      
      if (foundRow !== -1) {
        tsSheet.getRange(foundRow, 3).setValue(weekDataStr);
        tsSheet.getRange(foundRow, 4).setValue(new Date());
      } else {
        tsSheet.appendRow([email, weekId, weekDataStr, new Date()]);
      }
      
      return ContentService.createTextOutput(JSON.stringify({ status: "success" }))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
    if (action === "loadAll") {
      var email = payload.email.toLowerCase();
      var data = tsSheet.getDataRange().getValues();
      var userRecords = [];
      
      for (var i = 1; i < data.length; i++) {
        if (data[i][0].toLowerCase() === email) {
          userRecords.push(data[i][2]); // Returns raw data strings to parse on client
        }
      }
      
      return ContentService.createTextOutput(JSON.stringify({ status: "success", records: userRecords }))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
